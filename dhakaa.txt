// Helper: build a roughly rectangular polygon around a centre
// GeoJSON order: [lng, lat]
function box(lat: number, lng: number, dLat = 0.010, dLng = 0.012): number[][] {
  return [
    [lng - dLng, lat - dLat],
    [lng + dLng, lat - dLat],
    [lng + dLng, lat + dLat],
    [lng - dLng, lat + dLat],
    [lng - dLng, lat - dLat],  // close ring
  ];
}
