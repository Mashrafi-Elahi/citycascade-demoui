// FILE: data/types.ts
// PURPOSE: All shared TypeScript interfaces. Every city config
//          and every component imports from here.
//          This is the contract — change here, everything updates.
// ============================================================

export type DisasterType = 'FLOOD' | 'FIRE' | 'EARTHQUAKE' | 'DISEASE' | 'HEATWAVE';
export type RiskLevel = 'safe' | 'low' | 'moderate' | 'high' | 'critical';
export type ElevationType = 'low' | 'medium' | 'high';

export const ELEVATION_METERS: Record<ElevationType, number> = {
  low: 1.5,
  medium: 4.5,
  high: 8.0,
};

export const DRAINAGE_QUALITY: Record<string, number> = {
  poor: 0.2,
  moderate: 0.5,
  good: 0.8,
};

export interface Hospital {
  id: string;
  name: string;
  position: [number, number];
  capacity: number;
  currentLoad?: number; // 0-1, default 0.6
  phone: string;
  specializations?: string[];
}

export interface WaterSource {
  id: string;
  name: string;
  type: 'tap' | 'well' | 'reservoir' | 'river';
  position: [number, number];
  contamRisk?: number; // 0-1
}

export interface EmergencyService {
  id: string;
  name: string;
  phone: string;
  address?: string;
  position?: [number, number];
  type: 'police' | 'fire' | 'ambulance' | 'rescue' | 'utility' | 'ngo';
  coverage?: string[]; // zone ids
}

export interface EmergencyContacts {
  nationalEmergency: string;
  police: EmergencyService[];
  fire: EmergencyService[];
  ambulance: EmergencyService[];
  rescue: EmergencyService[];
  utility: EmergencyService[];
}

export interface Zone {
  id: string;
  name: string;
  center: [number, number];
  polygon: [number, number][];
  neighbors: string[];
  elevation: ElevationType;
  drainage: 'poor' | 'moderate' | 'good';
  population: number;
  buildingDensity?: number; // 0-1, used for fire spread
  soilType?: 'clay' | 'silt' | 'sand'; // used for earthquake amplification
  floodHistory?: 'none' | 'low' | 'moderate' | 'high';
  hospitals: Hospital[];
  waterSources: WaterSource[];
  roadNames: string[];
}

export interface ZoneImpact {
  zoneId: string;
  riskLevel: RiskLevel;
  intensity: number; // 0-1 cascade power at this zone
  // Disaster-specific (only one set will be populated per run)
  floodDepth?: number;        // meters
  fireSpreadPct?: number;     // 0-1
  groundShakingMMI?: number;  // 1-12 Modified Mercalli
  diseaseRisk?: number;       // 0-1
  heatStressIndex?: number;   // °C feels-like
  // Infrastructure (all 0-1)
  hospitalCapacityLoss: number;
  roadBlockedPct: number;
  waterContamRisk: number;
  displacedPopulation: number;
  // Routing output
  nearestHospitalId?: string;
  hospitalRoute?: [number, number][];
}

export interface CascadeResult {
  disasterType: DisasterType;
  epicenterZoneId: string;
  intensity: number;
  impacts: Record<string, ZoneImpact>;
  totalAffectedPop: number;
  criticalZoneIds: string[];
  blockedRoadNames: string[];
  overwhelmedHospitalIds: string[];
  timestamp: number;
}

export interface AIResponsePlan {
  summary: string;
  emergencyRouting: string[];
  hospitalAllocation: string[];
  publicHealth: string[];
  infrastructure: string[];
  evacuation: string[];
  contacts: string[];
}

// ── CITY CONFIG CONTRACT ──────────────────────────────────────
// TO ADD A NEW CITY: implement this interface in data/cities/yourcity.ts
// then register it in data/cityRegistry.ts. Nothing else changes.
export interface CityConfig {
  id: string;
  name: string;
  country: string;
  flag?: string;
  center: [number, number];
  zoom: number;
  timezone?: string;
  zones: Zone[];
  emergencyContacts: EmergencyContacts;
}
